use strict;

use Torus;


my $file = "torus-points-oo.pov";

`cat header.pov > $file`;


my $t = Torus->new(200, 30, 10, $file);

$t->init;

$t->tilt(0, 0, -45);

my $column_spacing = 200;
my $row_spacing = 470;

my $columns = 20;
my $rows    = 6;


for (my $j = 0; $j < $rows; $j++) {

  for (my $i = 0; $i < $columns; $i++) {

    $t->print;

    $t->move($column_spacing, 0, 0);

  }

  my $distance = $columns * $column_spacing * -1;

  $t->move($distance, 0, 0);

  $t->move(0, 0, $row_spacing);

}



my $s = Torus->new(200, 30, 10, $file);

$s->init;

$s->tilt(0, 0, 45);

$s->move($column_spacing / 2 + 15, 0, $row_spacing / 2);
$s->print;


for (my $j = 0; $j < $rows; $j++) {

  for (my $i = 0; $i < $columns; $i++) {

    $s->print;

    $s->move($column_spacing, 0, 0);

  }

  my $distance = $columns * $column_spacing * -1;

  $s->move($distance, 0, 0);

  $s->move(0, 0, $row_spacing);

}
